-- 1. Tabela de Categorias (Independente)
CREATE TABLE IF NOT EXISTS Categoria (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome TEXT NOT NULL
);

-- 2. Tabela de Clientes (Independente)
CREATE TABLE IF NOT EXISTS Cliente (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome TEXT NOT NULL,
    cpf TEXT UNIQUE NOT NULL
);

-- 3. Tabela de Fornecedores (Independente)
CREATE TABLE IF NOT EXISTS Fornecedor (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome_fantasia TEXT NOT NULL,
    razao_social TEXT,
    cnpj TEXT UNIQUE NOT NULL
);

-- 4. Tabela de Produtos (Depende de Categoria)
CREATE TABLE IF NOT EXISTS Produto (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome TEXT NOT NULL,
    preco_medio REAL DEFAULT 0,
    qtde_estoque REAL DEFAULT 0,
    valor_ultima_compra REAL DEFAULT 0,
    valor_ultima_venda REAL DEFAULT 0,
    id_categoria INTEGER NOT NULL,
    FOREIGN KEY (id_categoria) REFERENCES Categoria(id)
);

-- 5. Tabela de Compras (Capa da Compra - Depende de Fornecedor)
CREATE TABLE IF NOT EXISTS Compra (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    data_compra DATE NOT NULL,
    valor_total REAL DEFAULT 0,
    id_fornecedor INTEGER NOT NULL,
    FOREIGN KEY (id_fornecedor) REFERENCES Fornecedor(id)
);

-- 6. Tabela de Itens da Compra (Depende de Compra e Produto)
CREATE TABLE IF NOT EXISTS ItemCompra (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    id_compra INTEGER NOT NULL,
    id_produto INTEGER NOT NULL,
    quantidade REAL NOT NULL,
    valor_unitario REAL NOT NULL,
    FOREIGN KEY (id_compra) REFERENCES Compra(id),
    FOREIGN KEY (id_produto) REFERENCES Produto(id)
);

-- 7. Tabela de Vendas (Capa da Venda - Depende de Cliente)
CREATE TABLE IF NOT EXISTS Venda (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    data_venda DATE NOT NULL,
    valor_total REAL DEFAULT 0,
    id_cliente INTEGER NOT NULL,
    FOREIGN KEY (id_cliente) REFERENCES Cliente(id)
);

-- 8. Tabela de Itens da Venda (Depende de Venda e Produto)
CREATE TABLE IF NOT EXISTS ItemVenda (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    id_venda INTEGER NOT NULL,
    id_produto INTEGER NOT NULL,
    quantidade REAL NOT NULL,
    valor_unitario REAL NOT NULL,
    FOREIGN KEY (id_venda) REFERENCES Venda(id),
    FOREIGN KEY (id_produto) REFERENCES Produto(id)
);
